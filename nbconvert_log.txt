def cn_category(val) -> str:
    """Classify an absolute copy number value into CN=1, CN=2, or other."""
    if val is None or (isinstance(val, float) and np.isnan(val)):
        return 'unknown'
    if CN1_LO <= val < CN1_HI:
        return 'CN=1'
    if CN2_LO <= val < CN2_HI:
        return 'CN=2'
    return 'other'
